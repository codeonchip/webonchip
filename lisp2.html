LISP Interpreter (with Highlighting)

Type code below and press Run or Ctrl/⌘ + Enter.
; Sample program
(define square (lambda (x) (* x x)))
(define fact
  (lambda (n)
    (if (<= n 1)
        1
        (* n (fact (- n 1))))))

(begin
  (print (square 6))
  (print (fact 6))
  (print (let ((a 10) (b 20)) (+ a b)))
  (print (/ 3))     ; => 1/3
  (print (- 3))     ; => -3
)
